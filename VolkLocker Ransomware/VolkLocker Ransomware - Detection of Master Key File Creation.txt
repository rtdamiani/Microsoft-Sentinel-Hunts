DeviceFileEvents
| where TimeGenerated > ago(30d)
| where FolderPath contains @"C:\Users\" and FolderPath contains @"\AppData\Local\Temp"
| where FileName =~ "system_backup.key"
| where ActionType =~ "FileCreated"
| project TimeGenerated, DeviceName, FileName, FolderPath, SHA256, InitiatingProcessFileName, InitiatingProcessCommandLine
